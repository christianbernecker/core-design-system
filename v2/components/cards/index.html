IMPLEMENTATION GUIDE

HTML Structure

<!-- Standard Card -->
<div class="core-card">
    <div class="core-card-header">
        <h3 class="core-card-title">Card Title</h3>
        <div class="core-card-subtitle">Optional subtitle</div>
    </div>
    <div class="core-card-body">
        <p>Your card content goes here.</p>
    </div>
    <div class="core-card-footer">
        <button class="core-button secondary">Cancel</button>
        <button class="core-button primary">Confirm</button>
    </div>
</div>

<!-- Image Card -->
<div class="core-card">
    <img src="image.jpg" alt="Description" class="core-card-image">
    <div class="core-card-body">
        <h3 class="core-card-title">Image Card</h3>
        <p>Content below the image.</p>
    </div>
</div>

<!-- Full Picture Card -->
<div class="core-card full-picture" style="background-image: url('hero-image.jpg');">
    <div class="core-card-content">
        <h3 class="core-card-title">Hero Title</h3>
        <div class="core-card-body">
            <p>Full-height background image with overlay text.</p>
        </div>
        <div class="core-card-footer">
            <button class="core-button primary">Primary Action</button>
            <button class="core-button outline">Secondary</button>
        </div>
    </div>
</div>

Clickable Card JavaScript

// Make card clickable
document.querySelectorAll('.core-card.clickable').forEach(card => {
    card.addEventListener('click', () => {
        // Navigate to detail page
        window.location.href = card.dataset.href;
    });
});

// Card hover effects
document.querySelectorAll('.core-card').forEach(card => {
    card.addEventListener('mouseenter', () => {
        card.style.transform = 'translateY(-4px)';
    });
    
    card.addEventListener('mouseleave', () => {
        card.style.transform = 'translateY(0)';
    });
});

React/Next.js Component

// components/Card.tsx
import { ReactNode, HTMLAttributes } from 'react';

interface CardProps extends HTMLAttributes {
  title?: string;
  subtitle?: string;
  image?: string;
  footer?: ReactNode;
  variant?: 'default' | 'elevated' | 'compact';
  status?: 'success' | 'warning' | 'error' | 'info';
  horizontal?: boolean;
  clickable?: boolean;
  fullPicture?: boolean;
  backgroundImage?: string;
  children: ReactNode;
}

export const Card = ({
  title,
  subtitle,
  image,
  footer,
  variant = 'default',
  status,
  horizontal = false,
  clickable = false,
  fullPicture = false,
  backgroundImage,
  children,
  className = '',
  style,
  ...props
}: CardProps) => {
  const cardClasses = [
    'core-card',
    variant !== 'default' && variant,
    status,
    horizontal && 'horizontal',
    clickable && 'clickable',
    fullPicture && 'full-picture',
    className
  ].filter(Boolean).join(' ');

  const cardStyle = fullPicture && backgroundImage
    ? { ...style, backgroundImage: `url(${backgroundImage})` }
    : style;

  return (
    <div className={cardClasses} style={cardStyle} {...props}>
      {fullPicture ? (
        <div className="core-card-content">
          {title && <h3 className="core-card-title">{title}</h3>}
          <div className="core-card-body">{children}</div>
          {footer && <div className="core-card-footer">{footer}</div>}
        </div>
      ) : (
        <>
          {image && !horizontal && (
            <img src={image} alt="" className="core-card-image" />
          )}

          {horizontal ? (
            <div className="core-card-content">
              {(title || subtitle) && (
                <div className="core-card-header">
                  {title && <h3 className="core-card-title">{title}</h3>}
                  {subtitle && <div className="core-card-subtitle">{subtitle}</div>}
                </div>
              )}
              <div className="core-card-body">{children}</div>
              {footer && <div className="core-card-footer">{footer}</div>}
            </div>
          ) : (
            <>
              {(title || subtitle) && (
                <div className="core-card-header">
                  {title && <h3 className="core-card-title">{title}</h3>}
                  {subtitle && <div className="core-card-subtitle">{subtitle}</div>}
                </div>
              )}
              <div className="core-card-body">{children}</div>
              {footer && <div className="core-card-footer">{footer}</div>}
            </>
          )}
        </>
      )}
    </div>
  );
};

ACCESSIBILITY & BEST PRACTICES

Semantic HTML

  • Use appropriate heading levels
  • Ensure content flow is logical
  • Article or section elements when appropriate
  • Meaningful alt text for images

Keyboard Navigation

  • Interactive elements are tabbable
  • Clear focus indicators
  • Logical tab order
  • Enter/Space for clickable cards

Screen Readers

  • Clear content structure
  • Descriptive headings
  • Status announcements
  • Link purpose is clear

Visual Design

  • Sufficient contrast
  • Clear visual hierarchy
  • Consistent spacing
  • Responsive layout
