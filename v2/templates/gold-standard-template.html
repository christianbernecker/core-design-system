React/Next.js Integration

// components/Input.tsx
import { InputHTMLAttributes, ReactNode, forwardRef } from 'react';

interface InputProps extends InputHTMLAttributes {
  label?: string;
  helperText?: string;
  state?: 'default' | 'success' | 'error';
  size?: 'small' | 'medium' | 'large';
  icon?: ReactNode;
  floating?: boolean;
  glassmorphism?: boolean;
}

export const Input = forwardRef(({
  label,
  helperText,
  state = 'default',
  size = 'medium',
  icon,
  floating = false,
  glassmorphism = false,
  className = '',
  ...props
}, ref) => {
  const inputClasses = `
    core-input 
    ${size !== 'medium' ? size : ''} 
    ${state !== 'default' ? state : ''}
    ${glassmorphism ? 'glassmorphism' : ''}
    ${className}
  `.trim();

  const InputElement = (
    <input 
      ref={ref}
      className={inputClasses}
      {...props}
    />
  );

  return (
    <div className={`core-input-group ${floating ? 'core-input-floating' : ''}`}>
      {icon ? (
        <div className="core-input-with-icon">
          <div className="core-input-icon">{icon}</div>
          {InputElement}
        </div>
      ) : (
        InputElement
      )}
      {label && (
        <label className="core-input-label">{label}</label>
      )}
      {helperText && (
        <div className={`core-input-helper ${state !== 'default' ? state : ''}`}>
          {helperText}
        </div>
      )}
    </div>
  );
});

.core-input-with-icon Input container with icon support

.loading Loading state with animated indicator

Typography Scale Integration

Diese Input-Implementation nutzt die Typography Scale für 100%ige Konsistenz:

  • Small Inputs: var(--font-size-sm) (14px) - Kompakte UI-Elemente
  • Standard Inputs: var(--font-size-base) (16px) - CI-Standard
  • Large Inputs: var(--font-size-lg) (18px) - Wichtige Formulare
  • Labels: var(--font-size-sm) (14px) - Konsistente Beschriftung
  • Helper Text: var(--font-size-xs) (12px) - Zusatzinformationen
